## ?? Testing Guidelines

### Unit Tests
```csharp
[Fact]
public async Task CreateAsset_ValidAsset_ReturnsCreatedAsset()
{
    // Arrange
    var asset = new Asset 
    { 
        Name = "Test Asset",
        Cost = 100 
    };
    
    // Act
    var result = await _assetService.CreateAsync(asset);
    
    // Assert
    Assert.NotNull(result);
    Assert.Equal("Test Asset", result.Name);
}
```
